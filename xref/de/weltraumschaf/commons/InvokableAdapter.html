11  package de.weltraumschaf.commons;
12  
13  /**
14   * Abstract base implementation of an command line invokable.
15   *
16   * Extend this class in your client code.
17   *
18   * Example:
19   * <code>
20   *
21   * public class MyApp extends InvokableAdapter {
22   *
23   *      public static void main(final String[] args) {
24   *          InvokableAdapter.main(new App(args));
25   *      }
26   *
27   *      &#064;Override
28   *      public void execute() throws Exception {
29   *          // Your application code here.
30   *      }
31   *
32   * }
33   * </code>
34   *
35   * @author Sven Strittmatter <weltraumschaf@googlemail.com>
36   */
37  public abstract class InvokableAdapter implements Invokable {
38  
39      /**
40       * Copy of the command line arguments.
41       */
42      private final String[] args;
43  
44      /**
45       * I/O streams.
46       */
47      private IOStreams ioStreams;
48  
49      /**
50       * Container for shutdown {@link Runnable callbacks}.
51       */
52      private final ShutDownHook shutDownHooks = new ShutDownHook();
53  
54      /**
55       * Copies the command line arguments.
56       *
57       * @param args Command line arguments.
58       */
59      public InvokableAdapter(final String[] args) {
60          this.args = args.clone();
61      }
62  
63      /**
64       * Invokes {@link #main(de.weltraumschaf.commons.Invokable, de.weltraumschaf.commons.IOStreams)
65       * invokable} with default I/O.
66       *
67       * @param invokanle Implementation to invoke.
68       */
69      public static void main(Invokable invokanle) {
70          main(invokanle, IOStreams.newDefault());
71      }
72  
73      /**
74       * Inject the I/O streams to the invokable and then calls {@link Invokable#init()} and
75       * then {@link Invokable#execute()}.
76       *
77       * This method handles ell thrown {@link Exception} and calls {@link System#exit(int)}.
78       *
79       * @param invokanle Implementation to invoke.
80       * @param ioStreams I/O streams.
81       */
82      public static void main(final Invokable invokanle, final IOStreams ioStreams) {
83          invokanle.setIoStreams(ioStreams);
84  
85          try {
86              invokanle.init();
87              invokanle.execute();
88          } catch (Exception ex) {
89              ioStreams.println(ex.getMessage());
90              invokanle.exit(-1);
91          }
92  
93          invokanle.exit(0);
94      }
95  
96      /**
97       * Adds shutdown hook to runtime.
98       *
99       * @throws Exception Various exceptions from the runtime.
100      */
101     @Override
102     public final void init() throws Exception {
103         Runtime.getRuntime().addShutdownHook(shutDownHooks);
104     }
105 
106     /**
107      * Get the command line arguments.
108      *
109      * @return Returns copy of the arguments array.
110      */
111     public String[] getArgs() {
112         return args.clone();
113     }
114 
115     @Override
116     public IOStreams getIoStreams() {
117         return ioStreams;
118     }
119 
120     @Override
121     public void setIoStreams(IOStreams ioStreams) {
122         this.ioStreams = ioStreams;
123     }
124 
125     /**
126      * Get the shutdown hook object for registering callbacks executed on shutdown.
127      *
128      * @return Returns the shutdown object.
129      */
130     protected ShutDownHook getShutDownHooks() {
131         return shutDownHooks;
132     }
133 
134     @Override
135     public void exit(int status) {
136         System.exit(status);
137     }
138 
139 }
