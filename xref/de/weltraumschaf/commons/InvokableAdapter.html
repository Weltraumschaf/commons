12  package de.weltraumschaf.commons;
13  
14  import de.weltraumschaf.commons.system.DefaultExiter;
15  import de.weltraumschaf.commons.system.ExitCode;
16  import de.weltraumschaf.commons.system.Exitable;
17  import java.io.UnsupportedEncodingException;
18  import java.util.logging.Level;
19  import java.util.logging.Logger;
20  
21  /**
22   * Abstract base implementation of an command line invokable.
23   *
24   * Extend this class in your client code.
25   *
26   * Example:
27   * <code>
28   *
29   * public class MyApp extends InvokableAdapter {
30   *
31   *      public static void main(final String[] args) {
32   *          InvokableAdapter.main(new MyApp(args));
33   *      }
34   *
35   *      &#064;Override
36   *      public void execute() throws Exception {
37   *          // Your application code here.
38   *      }
39   *
40   * }
41   * </code>
42   *
43   * @author Sven Strittmatter <weltraumschaf@googlemail.com>
44   */
45  public abstract class InvokableAdapter implements Invokable {
46  
47      /**
48       * Copy of the command line arguments.
49       */
50      private final String[] args;
51  
52      /**
53       * Container for shutdown {@link Runnable callbacks}.
54       */
55      private final ShutDownHook shutDownHook;
56  
57      /**
58       * I/O streams.
59       */
60      private IOStreams ioStreams;
61  
62      /**
63       * Abstraction for {@link System#exit(int)}.
64       */
65      private Exitable exiter = new DefaultExiter();
66  
67      /**
68       * Runtime for registering callbacks.
69       */
70      private final Runtime runtime;
71  
72      /**
73       * Copies the command line arguments.
74       *
75       * @param args Command line arguments.
76       */
77      public InvokableAdapter(final String[] args) {
78          this(args, Runtime.getRuntime(), new ShutDownHook());
79      }
80  
81      /**
82       * Dedicated constructor.
83       *
84       * @param args Command line arguments.
85       * @param runtime Runtime for registering callbacks.
86       * @param shutDownHook Object to hold shutdown hooks.
87       */
88      public InvokableAdapter(final String[] args, final Runtime runtime, final ShutDownHook shutDownHook) {
89          this.args = args.clone();
90          this.runtime = runtime;
91          this.shutDownHook = shutDownHook;
92      }
93  
94      /**
95       * Invokes {@link #main(de.weltraumschaf.commons.Invokable, de.weltraumschaf.commons.IOStreams)
96       * invokable} with default I/O.
97       *
98       * @param invokable Implementation to invoke.
99       */
100     public static void main(final Invokable invokable) {
101         try {
102             main(invokable, IOStreams.newDefault());
103         } catch (UnsupportedEncodingException ex) {
104             Logger.getLogger(InvokableAdapter.class.getName()).log(Level.SEVERE, null, ex);
105             invokable.exit(-1);
106         }
107     }
108 
109     /**
110      * Inject the I/O streams to the invokable and then calls {@link Invokable#init()} and
111      * then {@link Invokable#execute()}.
112      *
113      * This method handles ell thrown {@link Exception} and calls {@link System#exit(int)}.
114      *
115      * @param invokable Implementation to invoke.
116      * @param ioStreams I/O streams.
117      */
118     public static void main(final Invokable invokable, final IOStreams ioStreams) {
119         main(invokable, ioStreams, false);
120     }
121 
122     /**
123      * Inject the I/O streams to the invokable and then calls {@link Invokable#init()} and
124      * then {@link Invokable#execute()}.
125      *
126      * This method handles ell thrown {@link Exception} and calls {@link System#exit(int)},
127      * and prints stack trace if <code>debug</code> is <tt>true</tt>.
128      *
129      * @param invokable implementation to invoke
130      * @param ioStreams I/O streams
131      * @param debug print stack trace if true
132      */
133     public static void main(final Invokable invokable, final IOStreams ioStreams, final boolean debug) {
134         invokable.setIoStreams(ioStreams);
135 
136         try {
137             invokable.init();
138             invokable.execute();
139         } catch (final Exception ex) {
140             ioStreams.errorln(ex.getMessage());
141 
142             if (debug) {
143                 ioStreams.printStackTrace(ex);
144             }
145 
146             if (ex instanceof ApplicationException) { // NOPMD Don't want to duplicate code for ERR out.
147                 invokable.exit(((ApplicationException) ex).getExitCode());
148             } else {
149                 invokable.exit(-1);
150             }
151         }
152 
153         invokable.exit(0);
154     }
155 
156     /**
157      * Adds shutdown hook to runtime.
158      */
159     @Override
160     public final void init() {
161         runtime.addShutdownHook(shutDownHook);
162     }
163 
164     /**
165      * Get the command line arguments.
166      *
167      * @return Returns copy of the arguments array.
168      */
169     public final String[] getArgs() {
170         return args.clone();
171     }
172 
173     @Override
174     public final IOStreams getIoStreams() {
175         return ioStreams;
176     }
177 
178     @Override
179     public final void setIoStreams(IOStreams ioStreams) {
180         this.ioStreams = ioStreams;
181     }
182 
183     /**
184      * Register a {@link Runnable callback} as shut down hook.
185      *
186      * @param callback On shutdown running callback.
187      */
188     public final void registerShutdownHook(final Runnable callback) {
189         shutDownHook.register(callback);
190     }
191 
192     @Override
193     public final void exit(int status) {
194         exiter.exit(status);
195     }
196 
197     @Override
198     public final void exit(ExitCode status) {
199         exiter.exit(status);
200     }
201 
202     @Override
203     public final void setExiter(final Exitable exiter) {
204         this.exiter = exiter;
205     }
206 
207 }
