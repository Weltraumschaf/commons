12  package de.weltraumschaf.commons.application;
13  
14  import de.weltraumschaf.commons.system.ExitCode;
15  import de.weltraumschaf.commons.system.Exitable;
16  
17  /**
18   * An invokable is an application object invokable by command line interface.
19   *
20   * @since 1.0.0
21   * @author Sven Strittmatter <weltraumschaf@googlemail.com>
22   */
23  public interface Invokable {
24  
25      /**
26       * Initializes the invokable.
27       */
28      void init();
29  
30      /**
31       * Template method for main application code.
32       *
33       * @throws Exception If something goes wrong.
34       */
35      void execute() throws Exception;
36  
37      /**
38       * Get stream object for I/O.
39       *
40       * @return Return always same instance until it was
41       *        {@link #setIoStreams(de.weltraumschaf.commons.application.IO) reset}.
42       */
43      IO getIoStreams();
44  
45      /**
46       * Set stream object for I/O.
47       *
48       * @param ioStreams Stream object.
49       */
50      void setIoStreams(IO ioStreams);
51  
52      /**
53       * Exits the invokable.
54       *
55       * @param status exit status.
56       */
57      void exit(ExitCode status);
58  
59      /**
60       * Exits the invokable.
61       *
62       * @param status exit status.
63       */
64      void exit(int status);
65  
66      /**
67       * Set an {@code exiter} to handle exit calls.
68       *
69       * @param exiter Object which handles exit calls.
70       */
71      void setExiter(Exitable exiter);
72  
73  }
