12  package de.weltraumschaf.commons.application;
13  
14  import de.weltraumschaf.commons.system.DefaultExiter;
15  import de.weltraumschaf.commons.system.ExitCode;
16  import de.weltraumschaf.commons.system.Exitable;
17  import java.io.UnsupportedEncodingException;
18  import java.util.logging.Level;
19  import java.util.logging.Logger;
20  
21  /**
22   * Abstract base implementation of an command line invokable.
23   *
24   * <p>
25   * Extend this class in your client code.
26   * </p>
27   *
28   * <p>
29   * Example:
30   * </p>
31   * <pre>{@code
32   *  public final class MyApp extends InvokableAdapter {
33   *
34   *      public MyApp(final String[] args) {
35   *          super(args);
36   *      }
37   *
38   *      public static void main(final String[] args) {
39   *          InvokableAdapter.main(new MyApp(args));
40   *      }
41   *
42   *      &#064;Override
43   *      public void execute() throws Exception {
44   *          registerShutdownHook(new Runnable() {
45   *
46   *              &#064;Override
47   *              public void run() {
48   *                  // Your shutdown code goes here.
49   *              }
50   *          });
51   *
52   *          // Your application code here.
53   *      }
54   *  }
55   * }</pre>
56   *
57   * @since 1.0.0
58   * @author Sven Strittmatter <weltraumschaf@googlemail.com>
59   */
60  public abstract class InvokableAdapter implements Invokable {
61  
62      /**
63       * Copy of the command line arguments.
64       */
65      private final String[] args;
66  
67      /**
68       * Container for shutdown {@link Runnable callbacks}.
69       */
70      private final ShutDownHook shutDownHook;
71  
72      /**
73       * I/O streams.
74       */
75      private IO ioStreams;
76  
77      /**
78       * Abstraction for {@link System#exit(int)}.
79       */
80      private Exitable exiter = new DefaultExiter();
81  
82      /**
83       * Runtime for registering callbacks.
84       */
85      private final Runtime runtime;
86  
87      /**
88       * Copies the command line arguments.
89       *
90       * @param args Command line arguments.
91       */
92      public InvokableAdapter(final String[] args) {
93          this(args, Runtime.getRuntime(), new ShutDownHook());
94      }
95  
96      /**
97       * Dedicated constructor.
98       *
99       * @param args Command line arguments.
100      * @param runtime Runtime for registering callbacks.
101      * @param shutDownHook Object to hold shutdown hooks.
102      */
103     public InvokableAdapter(final String[] args, final Runtime runtime, final ShutDownHook shutDownHook) {
104         this.args = args.clone();
105         this.runtime = runtime;
106         this.shutDownHook = shutDownHook;
107     }
108 
109     /**
110      * Invokes {@link #main(de.weltraumschaf.commons.application.Invokable, de.weltraumschaf.commons.application.IO)
111      * invokable} with default I/O.
112      *
113      * @param invokable Implementation to invoke.
114      */
115     public static void main(final Invokable invokable) {
116         try {
117             main(invokable, IOStreams.newDefault());
118         } catch (UnsupportedEncodingException ex) {
119             Logger.getLogger(InvokableAdapter.class.getName()).log(Level.SEVERE, null, ex);
120             invokable.exit(-1);
121         }
122     }
123 
124     /**
125      * Inject the I/O streams to the invokable and then calls {@link Invokable#init()} and then
126      * {@link Invokable#execute()}.
127      *
128      * This method handles ell thrown {@link Exception} and calls {@link System#exit(int)}.
129      *
130      * @param invokable Implementation to invoke.
131      * @param ioStreams I/O streams.
132      */
133     public static void main(final Invokable invokable, final IO ioStreams) {
134         main(invokable, ioStreams, false);
135     }
136 
137     /**
138      * Inject the I/O streams to the invokable and then calls {@link Invokable#init()} and then
139      * {@link Invokable#execute()}.
140      *
141      * This method handles ell thrown {@link Exception} and calls {@link System#exit(int)}, and prints stack trace if
142      * <code>debug</code> is <tt>true</tt>.
143      *
144      * @param invokable implementation to invoke
145      * @param ioStreams I/O streams
146      * @param debug print stack trace if true
147      */
148     public static void main(final Invokable invokable, final IO ioStreams, final boolean debug) {
149         invokable.setIoStreams(ioStreams);
150 
151         try {
152             invokable.init();
153             invokable.execute();
154         } catch (final Exception ex) {
155             ioStreams.errorln(ex.getMessage());
156 
157             if (debug) {
158                 ioStreams.printStackTrace(ex);
159             }
160 
161             if (ex instanceof ApplicationException) { // NOPMD Don't want to duplicate code for ERR out.
162                 invokable.exit(((ApplicationException) ex).getExitCode());
163             } else {
164                 invokable.exit(-1);
165             }
166         }
167 
168         invokable.exit(0);
169     }
170 
171     /**
172      * Adds shutdown hook to runtime.
173      */
174     @Override
175     public final void init() {
176         runtime.addShutdownHook(shutDownHook);
177     }
178 
179     /**
180      * Get the command line arguments.
181      *
182      * @return Returns copy of the arguments array.
183      */
184     public final String[] getArgs() {
185         return args.clone();
186     }
187 
188     @Override
189     public final IO getIoStreams() {
190         return ioStreams;
191     }
192 
193     @Override
194     public final void setIoStreams(final IO ioStreams) {
195         this.ioStreams = ioStreams;
196     }
197 
198     /**
199      * Register a {@link Runnable callback} as shut down hook.
200      *
201      * @param callback On shutdown running callback.
202      */
203     public final void registerShutdownHook(final Runnable callback) {
204         shutDownHook.register(callback);
205     }
206 
207     @Override
208     public final void exit(int status) {
209         exiter.exit(status);
210     }
211 
212     @Override
213     public final void exit(ExitCode status) {
214         exiter.exit(status);
215     }
216 
217     @Override
218     public final void setExiter(final Exitable exiter) {
219         this.exiter = exiter;
220     }
221 
222 }
