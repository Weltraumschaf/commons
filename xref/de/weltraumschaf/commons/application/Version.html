12  package de.weltraumschaf.commons.application;
13  
14  import java.io.IOException;
15  import java.io.InputStream;
16  import java.util.Properties;
17  
18  /**
19   * Reads the current version from a property file.
20   *
21   * <p>
22   * The property files only defines one property. Import is to switch resource filtering on in your POM
23   * for that properties file.
24   * </p>
25   * <pre>
26   *  version=1.0.0
27   * </pre>
28   *
29   * <p>
30   * This property file may be processed by a Maven filter to provide the verison from
31   * the pom.xml:
32   * </p>
33   * <pre>
34   *  version=${pom.version}
35   * </pre>
36   *
37   * <p>
38   * If you save the file in <kbd>src/main/resources/foo/bar/version.properties</kbd>
39   * you can use the version this way:
40   * </p>
41   *
42   * <pre>{@code
43   * Version version = new Version("/foo/bar/version.properties");
44   * version.load()
45   * println(version.getVersion);
46   * }</pre>
47   *
48   * @since 1.0.0
49   * @author Sven Strittmatter <weltraumschaf@googlemail.com>
50   */
51  public final class Version {
52  
53      /**
54       * Location of property file.
55       */
56      private final String propertyFileName;
57  
58      /**
59       * Indicates whether the properties are already loaded or not.
60       */
61      private boolean propertiesLoaded;
62  
63      /**
64       * Properties.
65       */
66      private final Properties properties = new Properties();
67  
68      /**
69       * Private constructor for singleton.
70       *
71       * @param propertyFileName Location of file, e.g. /de/weltraumschaf/commons/version.properties.
72       */
73      public Version(final String propertyFileName) {
74          super();
75          this.propertyFileName = propertyFileName;
76      }
77  
78      /**
79       * Opens the properties file and loads it.
80       *
81       * Only loads th property file once.
82       *
83       * @throws IOException On IO errors of the property file.
84       */
85      public void load() throws IOException {
86          if (!propertiesLoaded) {
87              loadImpl();
88              propertiesLoaded = true;
89          }
90      }
91  
92      /**
93       * Encapsulates the file loading.
94       *
95       * @throws IOException On IO errors of the property file.
96       */
97      void loadImpl() throws IOException {
98          InputStream in = null;
99  
100         try {
101             in = getClass().getResourceAsStream(propertyFileName);
102             properties.load(in);
103             in.close();
104         } finally {
105             if (null != in) {
106                 in.close();
107             }
108         }
109     }
110 
111     /**
112      * Get the version string.
113      *
114      * @return The version string.
115      */
116     public String getVersion() {
117         return properties.getProperty(PropertyNames.VERSION.toString(),
118                                       PropertyNames.VERSION.getDefaultValue());
119     }
120 
121     /**
122      * Returns the version string.
123      *
124      * @return Same as {@link #getVersion()}.
125      */
126     @Override
127     public String toString() {
128         return getVersion();
129     }
130 
131     /**
132      * Available properties in the file.
133      */
134     enum PropertyNames {
135 
136         /**
137          * Version property.
138          */
139         VERSION("version", "n/a");
140 
141         /**
142          * Property name.
143          */
144         private final String name;
145 
146         /**
147          * Default value.
148          */
149         private final String defaultValue;
150 
151         /**
152          * Initializes the enum with the property name.
153          *
154          * @param name The property name.
155          * @param defaultValue Fall back value.
156          */
157         PropertyNames(final String name, final String defaultValue) {
158             this.name = name;
159             this.defaultValue = defaultValue;
160         }
161 
162         /**
163          * Returns the property name.
164          *
165          * @return String containing the property name.
166          */
167         @Override
168         public String toString() {
169             return name;
170         }
171 
172         /**
173          * Get the fallback value.
174          *
175          * @return Return default value string.
176          */
177         public String getDefaultValue() {
178             return defaultValue;
179         }
180 
181     }
182 }
