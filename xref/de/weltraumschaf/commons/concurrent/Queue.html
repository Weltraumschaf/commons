13  package de.weltraumschaf.commons.concurrent;
14  
15  /**
16   * Interface for queues (FIFO).
17   *
18   * @since 1.0.0
19   * @param <E> type of stack entries
20   * @author Sven Strittmatter <weltraumschaf@googlemail.com>
21   */
22  public interface Queue<E> {
23  
24      /**
25       * Check if queue is empty.
26       *
27       * @return {@code true} if empty; else {@code false}
28       */
29      boolean isEmpty();
30  
31      /**
32       * Add element to the tail of queue.
33       *
34       * @param element added entry
35       */
36      void add(E element);
37  
38      /**
39       * Get element from head of queue.
40       *
41       * @return entry first element, {@code null} if queue is empty
42       */
43      E get();
44  
45  }
