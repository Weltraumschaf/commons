12  package de.weltraumschaf.commons.shell;
13  
14  import com.google.common.collect.Lists;
15  import java.util.List;
16  
17  /**
18   * Default parses implementation.
19   *
20   * Parsed grammar:
21   * <pre>
22   * imputline  = command { argument } .
23   * command    = keyword [ keyword ] .
24   * keyword    = character { character } .
25   * argument   = literal | number | string .
26   * literal    = alphanum { alphanum } .
27   * number     = digit { digit } .
28   * string     = '\'' alphanum { whitespace | alphanum } '\''
29                | '"' alphanum { whitespace | alphanum } '"' .
30   * alphanum   = character
31   *            | digit .
32   * character  = 'a' .. 'z'
33   *            | 'A' .. 'Z' .
34   * digit      = '0' .. '9' .
35   * whitespace = ' ' .
36   * </pre>
37   *
38   * @author Sven Strittmatter <weltraumschaf@googlemail.com>
39   */
40  class DefaultParser implements Parser {
41  
42      /**
43       * Tokenize the input line.
44       */
45      private final Scanner scanner;
46  
47      /**
48       * Verifies the parsed commands.
49       */
50      private final CommandVerifier verifier;
51  
52      private final LiteralCommandMap commandMap = null;
53  
54      /**
55       * Dedicated constructor.
56       *
57       * @param scanner to scan input line
58       * @param verifier verifies parsed commands
59       */
60      public DefaultParser(final Scanner scanner, final CommandVerifier verifier) {
61          super();
62          this.scanner = scanner;
63          this.verifier = verifier;
64      }
65  
66      /**
67       * Parses given input line.
68       *
69       * @param input line to parse
70       * @return recognized shell command
71       * @throws SyntaxException if, the parsed line has syntax errors
72       */
73      @Override
74      public ShellCommand parse(final String input) throws SyntaxException {
75          final List<Token> tokens = scanner.scan(input);
76          final Token commandtoken = tokens.get(0);
77  
78          if (TokenType.KEYWORD != commandtoken.getType()) {
79              throw new SyntaxException("Command expected as first input!");
80          }
81  
82          final MainCommandType command = commandMap.determineCommand(commandtoken);
83          // TODO If I use nul here, the command object should reflect this to prevent NPE.
84          SubCommandType subCommand = null;
85          int argumentBegin = 1;
86  
87          if (tokens.size() > 1) {
88              final Token secondToken = tokens.get(1);
89  
90              if (secondToken.getType() == TokenType.KEYWORD) {
91                  if (! commandMap.isSubCommand(secondToken)) {
92                      throw new SyntaxException(
93                              String.format("Command '%s' followed by bad keyword '%s' as sub command!",
94                                            commandtoken.getValue(), secondToken.getValue()));
95                  }
96                  ++argumentBegin;
97                  subCommand = commandMap.determineSubCommand(secondToken);
98              }
99          }
100 
101         List<Token> arguments;
102 
103         if (tokens.size() > argumentBegin) {
104             arguments = tokens.subList(argumentBegin, tokens.size());
105         } else {
106             arguments = Lists.newArrayList();
107         }
108 
109         final ShellCommand cmd = new ShellCommand(command, subCommand, arguments);
110         verifier.verifyCommand(cmd);
111         return cmd;
112     }
