19  /**
20   * Default parses implementation.
21   *
22   * Parsed grammar:
23   * <pre>
24   * imputline  = command { argument } .
25   * command    = keyword [ keyword ] .
26   * keyword    = character { character } .
27   * argument   = literal | number | string .
28   * literal    = alphanum { alphanum } .
29   * number     = digit { digit } .
30   * string     = '\'' alphanum { whitespace | alphanum } '\''
31                | '"' alphanum { whitespace | alphanum } '"' .
32   * alphanum   = character
33   *            | digit .
34   * character  = 'a' .. 'z'
35   *            | 'A' .. 'Z' .
36   * digit      = '0' .. '9' .
37   * whitespace = ' ' .
38   * </pre>
39   *
40   * @author Sven Strittmatter <weltraumschaf@googlemail.com>
41   */
42  class DefaultParser implements Parser {
43  
44      /**
45       * Required number of arguments for message command.
46       */
47      private static final int MESSAGE_ARGS_COUNT = 3;
48  
49      /**
50       * Tokenize the input line.
51       */
52      private final Scanner scanner;
53  
54      /**
55       * Dedicated constructor.
56       *
57       * @param scanner to scann input line
58       */
59      public DefaultParser(final Scanner scanner) {
60          super();
61          this.scanner = scanner;
62      }

64      /**
65       * Parses given input line.
66       *
67       * @param input line to parse
68       * @return recognized shell command
69       * @throws SyntaxException if, the parsed line has syntax errors
70       */
71      @Override
72      public ShellCommand parse(final String input) throws SyntaxException {
73          final List<Token> tokens = scanner.scan(input);
74          final Token commandtoken = tokens.get(0);
75  
76          if (TokenType.KEYWORD != commandtoken.getType()) {
77              throw new SyntaxException("Command expected as first input!");
78          }
79  
80          final MainType command = ShellCommand.determineCommand(commandtoken);
81          SubType subCommand = SubType.NONE;
82          int argumentBegin = 1;
83  
84          if (tokens.size() > 1) {
85              final Token secondToken = tokens.get(1);
86  
87              if (secondToken.getType() == TokenType.KEYWORD) {
88                  if (! ShellCommand.isSubCommand(secondToken)) {
89                      throw new SyntaxException(
90                              String.format("Command '%s' followed by bad keyword '%s' as sub command!",
91                                            commandtoken.getValue(), secondToken.getValue()));
92                  }
93                  ++argumentBegin;
94                  subCommand = ShellCommand.determineSubCommand(secondToken);
95              }
96          }
97  
98          List<Token> arguments;
99  
100         if (tokens.size() > argumentBegin) {
101             arguments = tokens.subList(argumentBegin, tokens.size());
102         } else {
103             arguments = Lists.newArrayList();
104         }
105 
106         final ShellCommand cmd = new ShellCommand(command, subCommand, arguments);
107         verifyCommand(cmd);
108         return cmd;
109     }

111     /**
112      * Verifies parsed command of consistency.
113      *
114      * Consistency checks are:
115      * - correct sub command type
116      * - correct number of arguments
117      *
118      * @param cmd command to verify
119      * @throws SyntaxException if, verification has failed
120      */
121     private void verifyCommand(final ShellCommand cmd) throws SyntaxException {
122         switch (cmd.getCommand()) {
123             case EXIT:
124             case HELP:
125             case RESET:
126                 if (cmd.getSubCommand() != SubType.NONE) {
127                     throw new SyntaxException(String.format("Command '%s' does not support subcommand '%s'!",
128                                                             cmd.getCommand(), cmd.getSubCommand()));
129                 }
130                 if (! cmd.getArguments().isEmpty()) {
131                     throw new SyntaxException(String.format("Command '%s' does not support arguments!",
132                                                             cmd.getCommand()));
133                 }
134                 break;
135             case NODE:
136                 verifyNodeCommand(cmd);
137                 break;
138             case MESSAGE:
139                 verifyMessageCommand(cmd);
140                 break;
141             case SAMPLE:
142                 verifySampleCommand(cmd);
143                 break;
144             case DUMP:
145                 verifyDumpCommand(cmd);
146                 break;
147             default:
148                 // Nothing to do here.
149         }
150     }
151 
152     /**
153      * Verify commands of main command type {@link MainType#NODE}.
154      *
155      * Consistency checks are:
156      * - correct number of arguments for sub command type
157      *
158      * @param cmd command to verify
159      * @throws SyntaxException if, wrong number of arguments or unsupported subcommand was parsed
160      */
161     private void verifyNodeCommand(final ShellCommand cmd) throws SyntaxException {
162         final int argumentCount = cmd.getArguments().size();
163 
164         if (cmd.getSubCommand() == SubType.NONE) {
165             throw new SyntaxException(String.format("Command '%s' must have sub command!", cmd.getCommand()));
166         }
167 
168         switch (cmd.getSubCommand()) {
169             case LIST:
170                 if (argumentCount != 0) {
171                     throw new SyntaxException(String.format("Command '%s %s' support no arguments!",
172                                                             cmd.getCommand(),
173                                                             cmd.getSubCommand()));
174                 }
175                 break;
176             case ADD:
177                 if (argumentCount > 1) {
178                     throw new SyntaxException(String.format("Command '%s %s' wants one or zero arguments!",
179                                                             cmd.getCommand(),
180                                                             cmd.getSubCommand()));
181                 }
182                 break;
183             case DEL:
184             case INFO:
185             case LISTEN:
186             case UNLISTEN:
187                 if (argumentCount != 1) {
188                     throw new SyntaxException(String.format("Command '%s %s' require one argument!",
189                                                             cmd.getCommand(),
190                                                             cmd.getSubCommand()));
191                 }
192                 break;
193             case CONNECT:
194             case DISCONNECT:
195                 if (argumentCount != 2) {
196                     throw new SyntaxException(String.format("Command '%s %s' require two arguments!",
197                                                             cmd.getCommand(),
198                                                             cmd.getSubCommand()));
199                 }
200                 break;
201             default:
202                 throw new SyntaxException(String.format("Command %s does not support subcommand %s!",
203                                                         cmd.getCommand(),
204                                                         cmd.getSubCommand()));
205 
206         }
207 
208     }

210     /**
211      * Verify commands of main command type {@link MainType#MESSAGE}.
212      *
213      * Consistency checks are:
214      * - correct number and type of arguments
215      *
216      * @param cmd command to verify
217      * @throws SyntaxException if, wrong number or type of arguments
218      */
219     private void verifyMessageCommand(final ShellCommand cmd) throws SyntaxException {
220         final List<Token> args = cmd.getArguments();
221 
222         if (args.size() != MESSAGE_ARGS_COUNT) {
223             throw new SyntaxException(String.format("Command '%s' requires 3 arguments!", cmd.getCommand()));
224         }
225 
226         if (args.get(0).getType() != TokenType.NUMBER) {
227             throw new SyntaxException(String.format("First argument of command '%s' must be a number!",
228                                                     cmd.getCommand()));
229         }
230 
231         if (args.get(1).getType() != TokenType.NUMBER) {
232             throw new SyntaxException(String.format("Second argument of command '%s' must be a number!",
233                                                     cmd.getCommand()));
234         }
235 
236         if (args.get(2).getType() != TokenType.STRING) {
237             throw new SyntaxException(String.format("Third argument of command '%s' must be a string!",
238                                                     cmd.getCommand()));
239         }
240     }
