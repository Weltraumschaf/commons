12  package de.weltraumschaf.commons.shell;
13  
14  import de.weltraumschaf.commons.guava.Lists;
15  import de.weltraumschaf.commons.token.Token;
16  import de.weltraumschaf.commons.token.TokenType;
17  import de.weltraumschaf.commons.validate.Validate;
18  import java.util.List;
19  
20  /**
21   * Default parses implementation.
22   * <p>
23   * Parsed grammar:
24   * </p>
25   * <pre>
26   * imputline  = command { argument } .
27   * command    = keyword [ keyword ] .
28   * keyword    = character { character } .
29   * argument   = literal | number | string | boolean.
30   * literal    = alphanum { alphanum } .
31   * number     = integer | float .
32   * integer    = [ sign ] digit { digit } .
33   * float      = [ sign ] digit { digit } '.' digit { digit } [ exponent ] .
34   * exponent   = ( 'e' | 'E' ) [ sign ] digit { digit } .
35   * boolean    = 'true' | 'false' .
36   * string     = '\'' alphanum { whitespace | alphanum } '\''
37                | '"' alphanum { whitespace | alphanum } '"' .
38   * alphanum   = character
39   *            | digit .
40   * character  = 'a' .. 'z'
41   *            | 'A' .. 'Z' .
42   * digit      = '0' .. '9' .
43   * sign       = '-' | '+' .
44   * whitespace = ' ' .
45   * </pre>
46   *
47   * @since 1.0.0
48   * @author Sven Strittmatter <weltraumschaf@googlemail.com>
49   */
50  class DefaultParser implements Parser {
51  
52      /**
53       * Tokenize the input line.
54       */
55      private final Scanner scanner;
56  
57      /**
58       * Verifies the parsed commands.
59       */
60      private final CommandVerifier verifier;
61  
62      /**
63       * Map used to determine command keywords.
64       */
65      private final LiteralCommandMap commandMap;
66  
67      /**
68       * Dedicated constructor.
69       *
70       * @param scanner must not be {@code null}
71       * @param verifier must not be {@code null}
72       * @param commandMap must not be {@code null}
73       */
74      public DefaultParser(final Scanner scanner, final CommandVerifier verifier, final LiteralCommandMap commandMap) {
75          super();
76          this.scanner = Validate.notNull(scanner, "scanner");
77          this.verifier = Validate.notNull(verifier, "verifier");
78          this.commandMap = Validate.notNull(commandMap, "commandMap");
79      }
80  
81      /**
82       * Parses given input line.
83       *
84       * @param input must not be {@code null}
85       * @return never {@code null}
86       * @throws SyntaxException if the parsed line has syntax errors
87       */
88      @Override
89      public ShellCommand parse(final String input) throws SyntaxException {
90          final List<Token> tokens = scanner.scan(input);
91          final Token commandtoken = tokens.get(0);
92  
93          if (TokenType.KEYWORD != commandtoken.getType()) {
94              throw new SyntaxException("Command expected as first input!");
95          }
96  
97          final MainCommandType command = commandMap.determineCommand(commandtoken);
98          SubCommandType subCommand = commandMap.getDefaultSubCommand();
99          int argumentBegin = 1;
100 
101         if (tokens.size() > 1) {
102             final Token secondToken = tokens.get(1);
103 
104             if (secondToken.getType() == TokenType.KEYWORD) {
105                 final Token keyword = secondToken;
106 
107                 if (! commandMap.isSubCommand(keyword)) {
108                     throw new SyntaxException(
109                             String.format("Command '%s' followed by bad keyword '%s' as sub command!",
110                                           commandtoken.asString(), secondToken.asString()));
111                 }
112 
113                 ++argumentBegin;
114                 subCommand = commandMap.determineSubCommand(keyword);
115             }
116         }
117 
118         final List<Token> arguments;
119 
120         if (tokens.size() > argumentBegin) {
121             arguments = tokens.subList(argumentBegin, tokens.size());
122         } else {
123             arguments = Lists.newArrayList();
124         }
125 
126         final ShellCommand cmd = new ShellCommand(command, subCommand, arguments);
127         verifier.verifyCommand(cmd);
128         return cmd;
129     }
130 
131 }
