12  package de.weltraumschaf.commons.shell;

14  import com.google.common.base.Objects;
15  import com.google.common.collect.Lists;
16  import com.google.common.collect.Maps;

20  /**
21   * Describes a parsed command of the interactive shell.
22   *
23   * Commands consist always of a command and may have a subcommand.
24   * This type is immutable.
25   *
26   * @author Sven Strittmatter <weltraumschaf@googlemail.com>
27   */
28  public class ShellCommand {
29  
30      /**
31       * Map the literal command string to corresponding type enum.
32       */
33      private static final Map<String, MainType> COMMANDS = Maps.newHashMap();
34  
35      /**
36       * Map the literal sub command string to corresponding type enum.
37       */
38      private static final Map<String, SubType> SUB_COMMANDS = Maps.newHashMap();
39  
40      static {
41          for (final MainType t : MainType.values()) {
42              COMMANDS.put(t.toString(), t);
43          }
44          for (final SubType t : SubType.values()) {
45              if (t == SubType.NONE) {
46                  continue; // Ignore to do not recognize empty strings in #isSubCommand().
47              }
48              SUB_COMMANDS.put(t.toString(), t);
49          }
50      }
51  
52      /**
53       * Obligatory main command.
54       */
55      private final MainType command;
56  
57      /**
58       * Optional sub command.
59       *
60       * If the main type does not support sub commands this will be {@link SubType#NONE}.
61       */
62      private final SubType subCommand;
63  
64      /**
65       * Optional arguments.
66       *
67       * If the command has no arguments this will be an empty list.
68       */
69      private final List<Token> arguments;
70  
71      /**
72       * Constructor for commands with no supported sub commands.
73       *
74       * @param command main command
75       * @param arguments command arguments, may be an empty list
76       */
77      public ShellCommand(final MainType command, final List<Token> arguments) {
78          this(command, SubType.NONE, arguments);
79      }
80  
81      /**
82       * Dedicated constructor.
83       *
84       * @param command main command
85       * @param subCommand sub command
86       * @param arguments command arguments, may be an empty list
87       */
88      public ShellCommand(final MainType command, final SubType subCommand, final List<Token> arguments) {
89          super();
90          this.command    = command;
91          this.subCommand = subCommand;
92          this.arguments  = Lists.newArrayList(arguments); // Defense copy
93      }
94  
95      /**
96       * Get command main type.
97       *
98       * @return main type of command
99       */
100     public MainType getCommand() {
101         return command;
102     }
103 
104     /**
105      * Get optional command sub type.
106      *
107      * If the main type does not support any sub type {@link SubType#NONE} will be returned.
108      *
109      * @return sub type of command
110      */
111     public SubType getSubCommand() {
112         return subCommand;
113     }
114 
115     /**
116      * Get arguments.
117      *
118      * This method will always return a list. If no arguments are present an empty list will be returned.
119      *
120      * @return Returns a defense copy.
121      */
122     public List<Token> getArguments() {
123         return Lists.newArrayList(arguments); // Defense copy
124     }
125 
126     /**
127      * Determines if the string literal value of the token is a main command.
128      *
129      * TODO Maybe remove this.
130      *
131      * @param t token to check
132      * @return true if the token is a command else false
133      */
134     static boolean isCommand(final Token<String> t) {
135         return COMMANDS.containsKey(t.getValue());
136     }
137 
138     /**
139      * Determines the appropriate main command type for given string token.
140      *
141      * You should check with {@link #isCommand(de.weltraumschaf.neuron.shell.Token)} before invocation.
142      * Determining a non command token string will result in an exception.
143      *
144      * @param t token to check
145      * @return command main type
146      * // CHECKSTYLE:OFF
147      * @throws IllegalArgumentException if, token is not a main command
148      * // CHECKSTYLE:ON
149      */
150     static MainType determineCommand(final Token<String> t) {
151         if (! isCommand(t)) {
152             throw new IllegalArgumentException(String.format("'%s' is not a command!", t.getValue()));
153         }
154         return COMMANDS.get(t.getValue());
155     }
156 
157     /**
158      * Determines if the string literal value of the token is a sub command.
159      *
160      * TODO Maybe remove this.
161      *
162      * @param t token to check
163      * @return true if the token is a sub command else false
164      */
165     static boolean isSubCommand(final Token<String> t) {
166         return SUB_COMMANDS.containsKey(t.getValue());
167     }
168 
169     /**
170      * Determines the appropriate sub command type for given string token.
171      *
172      * You should check with {@link #isSubCommand(de.weltraumschaf.neuron.shell.Token)} before invocation.
173      * Determining a non sub command token string will result in an exception.
174      *
175      * @param t token to check
176      * @return command sub type
177      * // CHECKSTYLE:OFF
178      * @throws IllegalArgumentException if, token is not a sub command
179      * // CHECKSTYLE:ON
180      */
181     static SubType determineSubCommand(final Token<String> t) {
182         if (! isSubCommand(t)) {
183             throw new IllegalArgumentException(String.format("'%s' is not a sub command!", t.getValue()));
184         }
185         return SUB_COMMANDS.get(t.getValue());
186     }
187 
188     @Override
189     public String toString() {
190         return Objects.toStringHelper(this)
191                       .add("mainCommand", command)
192                       .add("subCommand", subCommand)
193                       .add("arguments", arguments)
194                       .toString();
195     }

204     /**
205      * Enumerates the available commands.
206      */
207     public enum MainType implements Type {
208 
209         /** Help command. */
210         HELP("help"),
211         /** Reset command. */
212         RESET("reset"),
213         /** Exit command. */
214         EXIT("exit"),
215         /** Node command. */
216         NODE("node"),
217         /** Message command. */
218         MESSAGE("message"),
219         /** Dump command. */
220         DUMP("dump"),
221         /** Sample command. */
222         SAMPLE("sample");
223 
224         /**
225          * Literal command string used in shell.
226          */
227         private final String name;
228 
229         /**
230          * Initialize name.
231          *
232          * @param name literal shell command string
233          */
234         private MainType(final String name) {
235             this.name = name;
236         }
237 
238         @Override
239         public String toString() {
240             return name;
241         }
242 
243     }

245     /**
246      * Enumerates the optional subcommands.

248     public enum SubType implements Type {

250         NONE(""),

252         ADD("add"),
253         /** Del sub command for node command. */
254         DEL("del"),
255         /** Connect sub command for node command. */
256         CONNECT("connect"),
257         /** Disconnect sub command for node command. */
258         DISCONNECT("disconnect"),
259         /** List sub command for node command. */
260         LIST("list"),
261         /** Info sub command for node command. */
262         INFO("info"),
263         /** Listen sub command for node command. */
264         LISTEN("listen"),
265         /** Unlisten sub command for node command. */
266         UNLISTEN("unlisten"),
267         /** Dot subcommand for dump command. */
268         DOT("dot"),
269         /** Tree subcommand for dump command. */
270         TREE("tree"),
271         BITREE("bitree");
272 
273         /**
274          * Literal command string used in shell.

290         }
