13  package de.weltraumschaf.commons.string;
14  
15  /**
16   * Various string utility methods.
17   *
18   * @since 1.0.1
19   * @author Sven Strittmatter <weltraumschaf@googlemail.com>
20   */
21  public final class Strings {
22  
23      /**
24       * Hidden for pure static utility class.
25       */
26      private Strings() {
27          super();
28          throw new UnsupportedOperationException("Do not call by reflection!");
29      }
30  
31      /**
32       * Trims passed in string and does not throw {@link NullPointerException} if passed in string is {@code null}.
33       *
34       * @param fileName may be {@code null}
35       * @return never {@code null}
36       */
37      public static String nullAwareTrim(final String fileName) {
38          return null == fileName ? "" : fileName.trim();
39      }
40  
41  }
