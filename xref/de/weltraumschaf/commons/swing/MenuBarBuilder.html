12  package de.weltraumschaf.commons.swing;
13  
14  import javax.swing.JMenuBar;
15  
16  /**
17   * This builder offers a internal DSL for creating menu bars.
18   * <p>
19   * Example:
20   * </p>
21   * <pre>{@code
22   * final JMenuBar menubar = MenuBarBuilder.builder()
23   *            .menu("File")
24   *                .item("Open")
25   *                    .addListener(new Listener())
26   *                .end()
27   *                .separator()
28   *                .item("Save")
29   *                    .addListener(new Listener())
30   *                .end()
31   *            .end()
32   *            .menu("Edit")
33   *                .item("foo")
34   *                .end()
35   *            .end()
36   *            .menu("View")
37   *                .item("bar")
38   *                .end()
39   *            .end()
40   *            .menu("Window")
41   *                .item("baz")
42   *                .end()
43   *            .end()
44   *        .create();
45   * }</pre>
46   *
47   * @since 1.0.0
48   * @author Sven Strittmatter <weltraumschaf@googlemail.com>
49   */
50  public final class MenuBarBuilder {
51  
52      /**
53       * The built menu bar.
54       */
55      private final JMenuBar menuBar = new JMenuBar();
56  
57      /**
58       * Use {@link #builder()} instead.
59       */
60      private MenuBarBuilder() {
61          super();
62      }
63  
64      /**
65       * Creates a new builder instance.
66       *
67       * @return Always a new instance.
68       */
69      public static MenuBarBuilder builder() {
70          return new MenuBarBuilder();
71      }
72  
73      /**
74       * Returns the built menu bar.
75       *
76       * @return Returns always the same instance per builder.
77       */
78      public JMenuBar create() {
79          return menuBar;
80      }
81  
82      /**
83       * Creates a menu entry in menu bar.
84       *
85       * @param name Name of the menu.
86       * @return A menu builder.
87       */
88      public MenuBuilder menu(final String name) {
89          final MenuBuilder builder = new MenuBuilder(name, this);
90          menuBar.add(builder.getMenu());
91          return builder;
92      }
93  
94  }
