12  package de.weltraumschaf.commons.swing;
13  
14  import java.net.URL;
15  import javax.swing.Icon;
16  import javax.swing.ImageIcon;
17  import javax.swing.JToolBar;
18  
19  /**
20   * This builder offers a internal DSL for creating tool bars.
21   * <p>
22   * Example:
23   * </p>
24   * <pre>{@code
25   * final JToolBar toolbar = ToolBarBuilder.builder()
26   *            .button("/de/weltraumschaf/swing/folder_16x16.gif")
27   *                .toolTipText("Open an existing document.")
28   *                .addListener(new Listener())
29   *            .end()
30   *            .button("/de/weltraumschaf/swing/disk_16x16.gif")
31   *                .toolTipText("Save current document.")
32   *                .addListener(new Listener())
33   *            .end()
34   *            .button("/de/weltraumschaf/swing/page_16x16.gif")
35   *                .toolTipText("Create a new document.")
36   *                .addListener(new Listener())
37   *            .end()
38   *            .create();
39   * }</pre>
40   *
41   * @since 1.0.0
42   * @author Sven Strittmatter <weltraumschaf@googlemail.com>
43   */
44  public final class ToolBarBuilder {
45  
46      /**
47       * The built tool bar.
48       */
49      final JToolBar toolbar = new JToolBar();
50  
51      /**
52       * Use {@link #builder()} instead.
53       */
54      private ToolBarBuilder() {
55          super();
56      }
57  
58      /**
59       * Creates a new builder instance.
60       *
61       * @return Always a new instance.
62       */
63      public static ToolBarBuilder builder() {
64          return new ToolBarBuilder();
65      }
66  
67      /**
68       * Creates a button in the tool bar.
69       *
70       * @param iconResource Path to the button's icon resource.
71       * @return A button builder.
72       */
73      public ToolBarButtonBuilder button(final URL iconResource) {
74          return button(new ImageIcon(iconResource));
75      }
76  
77      /**
78       * Creates a button in the tool bar.
79       *
80       * @param icon Icon of the button.
81       * @return A button builder.
82       */
83      public ToolBarButtonBuilder button(final Icon icon) {
84          final ToolBarButtonBuilder builder = new ToolBarButtonBuilder(icon, this);
85          toolbar.add(builder.getButton());
86          return builder;
87      }
88  
89      /**
90       * Returns the built tool bar.
91       *
92       * @return Returns always the same instance per builder.
93       */
94      public JToolBar create() {
95          return toolbar;
96      }
97  
98  }
