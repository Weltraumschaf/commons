12  package de.weltraumschaf.commons.swing;
13  
14  import java.awt.event.ActionListener;
15  import javax.swing.Icon;
16  import javax.swing.JButton;
17  
18  /**
19   * Builder to create a tool bar button.
20   * <p>
21   * You should not use this class directly. Instead use {@link ToolBarBuilder}.
22   * </p>
23   *
24   * @since 1.0.0
25   * @author Sven Strittmatter <weltraumschaf@googlemail.com>
26   */
27  public final class ToolBarButtonBuilder {
28  
29      /**
30       * The built button.
31       */
32      private final JButton button;
33  
34      /**
35       * The tool bar builder which creates the tool bar to which button will belong.
36       */
37      private final ToolBarBuilder parent;
38  
39      /**
40       * Package private constructor.
41       *
42       * This sub builder is constructed by the {@link ToolBarBuilder}.
43       *
44       * @param image Image of the button.
45       * @param parent Constructing tool bar builder.
46       */
47      ToolBarButtonBuilder(final Icon image, final ToolBarBuilder parent) {
48          this.button = new JButton(image);
49          this.parent = parent;
50      }
51  
52      /**
53       * Get the built button.
54       *
55       * @return Always return the same instance per builder.
56       */
57      JButton getButton() {
58          return button;
59      }
60  
61      /**
62       * Set tool tip text of the button.
63       *
64       * @param text Tool tip text.
65       * @return The builder itself.
66       */
67      public ToolBarButtonBuilder toolTipText(final String text) {
68          button.setToolTipText(text);
69          return this;
70      }
71  
72      /**
73       * Add a listener to the button.
74       *
75       * @param listener Action listener to handle events.
76       * @return Returns the builder itself.
77       */
78      public ToolBarButtonBuilder addListener(final ActionListener listener) {
79          button.addActionListener(listener);
80          return this;
81      }
82  
83      /**
84       * Call if ready with building the button.
85       *
86       * @return The parent tool bar builder.
87       */
88      public ToolBarBuilder end() {
89          return parent;
90      }
91  
92  }
